namespace = noble_auto_marry

noble_auto_marry.1 = {
	type = country_event
	title = noble_auto_marry.1.t
	desc = noble_auto_marry.1.d
	
	trigger = {
		is_ai = no
		NOT = { has_variable = hide_marriage_events }
		any_character = {
			is_married = no
			is_female = no
			OR = {
				has_estate = estate_type:nobles_estate 
				AND = {
					has_estate = estate_type:crown_estate 
					NOT = { is_eligible_heir_baseline = root }
				}
			}
			age_in_years > 18
			age_in_years < 25
		}
		any_character = {
			is_married = no
			is_female = yes
			OR = {
				has_estate = estate_type:nobles_estate 
				AND = {
					has_estate = estate_type:crown_estate 
					NOT = { is_eligible_heir_baseline = root }
				}
			}
			age_in_years > 18
			age_in_years < 25
		}
	}
	
	immediate = {
		random_character = {
			limit = {
				is_married = no
				is_female = no
				OR = {
					has_estate = estate_type:nobles_estate 
					AND = {
						has_estate = estate_type:crown_estate 
						NOT = { is_eligible_heir_baseline = root }
					}
				}
				age_in_years > 18
				age_in_years < 25
			}
			save_scope_as = target_male
		}
		random_character = {
			limit = {
				is_married = no
				is_female = yes
				NOT = {
					is_close_relative = scope:target_male
				}		
				OR = {
					has_estate = estate_type:nobles_estate 
					AND = {
						has_estate = estate_type:crown_estate 
						NOT = { is_eligible_heir_baseline = root }
					}
				}
				age_in_years > 18
				age_in_years < 25
			}
			save_scope_as = target_female
		}
	}
	option = {
		name = noble_auto_marry.1.a
		scope:target_male ?= { marry_character = scope:target_female }
	}
	option = {
		name = noble_auto_marry.1.b
		scope:target_male ?= { marry_character = scope:target_female }
		custom_tooltip =  { text = hide_marriage_events.tt }
		set_variable = hide_marriage_events
	}
}

					name = auto_marry_reminder_sent
				}
			}

			random_character = {

					is_married = no
					AND = {
