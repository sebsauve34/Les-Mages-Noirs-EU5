namespace = german_events_nacomp

german_events_nacomp.1 = {
	type = country_event
	title = german_events_nacomp.1.title
	desc = german_events_nacomp.1.desc
    fire_only_once = yes
	dynamic_historical_event = {
        tag = HES
		from = 1337.1.1
		to = 1400.12.30
		monthly_chance = 100
	}


    trigger = {
		
		culture = { has_culture_group = culture_group:german_group}
		is_human = yes
		total_population <300 
        NOT = {
           has_or_had_tag = LBV
           has_or_had_tag = UBV
        }
	}


	immediate = {
		event_illustration_estate_effect = { foreground = estate_type:nobles_estate background = estate_type:nobles_estate }
	}

	option = {
		name = german_events_nacomp.1.a
		add_country_modifier = { modifier = german_minor_expansion years = 30 mode = add_and_extend }
	}
}
