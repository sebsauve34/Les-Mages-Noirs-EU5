get_nobles_auto_marry_eligibility_settings = {
	type = country
	
	text = {
		trigger = {
			NOT = {
				OR = {
					has_variable = nobles_auto_marry_selective_marriage
					has_variable = nobles_auto_marry_dynastic_preservation
				}
			}
		}
		localization_key = unrestricted_noble_auto_marry
	}
	
	text = {
		trigger = {
			has_variable = nobles_auto_marry_selective_marriage
		}
		localization_key = selective_noble_auto_marry

	text = {
		trigger = {
			has_variable = nobles_auto_marry_dynastic_preservation
		}
		localization_key = dynastic_preservation_noble_auto_marry
	}
}
