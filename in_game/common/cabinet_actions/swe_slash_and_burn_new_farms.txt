swe_slash_and_burn_new_farms = {
	ability = adm
	icon = pop_promote_action
	allow_multiple = yes
	
	potential = {
		OR = {
			tag = SWE
			tag = FIN
		}
	}
	
	
	select_trigger = {
		looking_for_a = province
		source = actor
		target_flag = target
		name = "swe_slash_and_burn_new_farms_select_province"
		none_available_msg_key = "swe_slash_and_burn_new_farms_no_provinces"
		column = {
			data = name
		}
		column = {
			data = population
		}
		visible = {
			province_average_development < 25
			population < 15
			OR = {
				area = area:finland_area
				area = area:norrland_area
				area = area:kola_area
				area = area:karelia_area
			}
			province_definition != province_definition:finland_province
			province_definition != province_definition:nyland_province
			
			scope:actor = {
				not = {
					any_cabinet = {
						has_cabinet_action = yes
						cabinet_action = cabinet_action:swe_slash_and_burn_new_farms
						interaction_target:target ?= root
					}
				}
			}
		}
		map_color = {
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = province.province_average_development
			}
		}
	}

	is_finished = {
		scope:target ?= {
			population > 20
		}
	}
	
	location_modifier = {
		local_population_growth = 0.05
		local_monthly_development = 0.02
	}

	map_marker = {
		show_for_owner = yes
	}
}
